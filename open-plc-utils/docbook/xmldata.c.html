#ifndef XMLDATA_SOURCE
#define XMLDATA_SOURCE

#include <string.h>

#include "../nodes/node.h"

struct node const * xmldata (struct node const * node)

{
	if (node)
	{
		node = node->below;
	}
	while (node)
	{
		if (node->type == NODE_DATA)
		{
			break;
		}
		node=node->after;
	}
	return (node);
}

#endif
